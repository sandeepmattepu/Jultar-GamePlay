277  protected virtual void observeRoomForPlayerLeavingAndEntering()
278  {
279  SMLeftOrJoinedPlayersData modifiedData = dataAboutPlayersLeftOrJoined ();
280  PhotonPlayer tempPlayer;
281 
282  if(modifiedData != null && PhotonNetwork.isMasterClient)
283  {
284  if(modifiedData.HasPlayerJoined)
285  {
286  for(int i = 0; i < modifiedData.PlayersIdsWhoLeftOrJoined.Length; i++)
287  {
288  tempPlayer = PhotonPlayer.Find (modifiedData.PlayersIdsWhoLeftOrJoined [i]);
289  if(tempPlayer != null)
290  {
291  playersIdAndName.Add (tempPlayer.ID, tempPlayer.NickName);
292  playerIdAndDeaths.Add (tempPlayer.ID, 0);
293  playersIdAndScore.Add (tempPlayer.ID, 0);
294  }
295  }
296  }
297  else
298  {
299  for(int i = 0; i < modifiedData.PlayersIdsWhoLeftOrJoined.Length; i++)
300  {
301  playersIdAndName.Remove(modifiedData.PlayersIdsWhoLeftOrJoined[i]);
302  playerIdAndDeaths.Remove(modifiedData.PlayersIdsWhoLeftOrJoined[i]);
303  playersIdAndScore.Remove(modifiedData.PlayersIdsWhoLeftOrJoined[i]);
304  }
305  }
306 
307  playersAtStartOfGame = PhotonNetwork.playerList;
308  dataTransmitter.sendSignalToAllToRefreshData ();
309 
310  if(OnPlayerJoinedOrLeft != null)
311  {
312  OnPlayerJoinedOrLeft ();
313  }
314  }
315  }

321  protected SMLeftOrJoinedPlayersData dataAboutPlayersLeftOrJoined()
322  {
323  bool joinedOrLeft = true;
324  if(playersAtStartOfGame.Length != PhotonNetwork.playerList.Length) // For optimizing
325  {
326  List<int> playerModifiedData = new List<int>();
327  List<int> currentPlayersID = new List<int> ();
328 
329  for(int i = 0; i < PhotonNetwork.playerList.Length; i++)
330  {
331  currentPlayersID.Add (PhotonNetwork.playerList [i].ID);
332  }
333 
334  // When players leave
335  if(playersAtStartOfGame.Length > PhotonNetwork.playerList.Length)
336  {
337  joinedOrLeft = false; // false means left
338 
339  for(int i = 0; i < playersAtStartOfGame.Length; i++)
340  {
341  if(!currentPlayersID.Contains(playersAtStartOfGame[i].ID))
342  {
343  playerModifiedData.Add (playersAtStartOfGame [i].ID);
344  }
345  }
346  }
347  // Someone joined game
348  else if(playersAtStartOfGame.Length < PhotonNetwork.playerList.Length)
349  {
350  joinedOrLeft = true; // true means joined
351 
352  for(int i = 0; i < playersAtStartOfGame.Length; i++)
353  {
354  currentPlayersID.Remove (playersAtStartOfGame [i].ID);
355  }
356  playerModifiedData = currentPlayersID;
357  }
358 
359  return new SMLeftOrJoinedPlayersData (playerModifiedData.ToArray (), joinedOrLeft);
360  }
361 
362  return null;
363  }
