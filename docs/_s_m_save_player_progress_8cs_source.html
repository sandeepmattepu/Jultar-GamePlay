204  public static int calculateTotalXpRequiredForNextLevel(int currentLevel)
205  {
206  int factor = currentLevel / 10;
207  int xpRequiredUptoNextTenLevels = 700000 + (factor * 50000);
208  xpRequiredUptoNextTenLevels = factor >= 8 ? (2000000) : xpRequiredUptoNextTenLevels;
209  int xpRequiredForEachLevel = (factor == 0 ? (xpRequiredUptoNextTenLevels / 9) : (xpRequiredUptoNextTenLevels / 10));
210  return xpRequiredForEachLevel;
