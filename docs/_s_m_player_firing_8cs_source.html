457  public void performFiring()
458  {
459  if(ammoDetails.bulletsLeft > 0 && (isReloadingFinished && isThrowingGrenadeFinished))
460  {
461  if(timer >= secondsPerBullet)
462  {
463  ownerClientBeganFiring = true;
464  ammoDetails.bulletsLeft -= 1;
465  totalNumberOfBullets -= 1;
466  timer = 0.0f;
467  // play firing audio
468  if(audioClip != null)
469  {
470  if(!audioSource.isPlaying)
471  {
472  audioSource.Stop ();
473  }
474  audioSource.Play ();
475 
476  }
477  // Show particle effects
478  if(gunTip.GetComponent<ParticleSystem>().isPlaying)
479  {
480  gunTip.GetComponent<ParticleSystem> ().Stop ();
481  }
482  gunTip.GetComponent<ParticleSystem> ().Play ();
483 
484  // Show line renderer and cast physics raycast
485  createPhysicsRayCastAndLineRenderer();
486  // Show light when gun is fired
487  lightFromGunFire.enabled = true;
488  setGunPositionWhileFiring (amountOfGunKickBackFiring, true); // Move the gun back
489  }
490  else
491  {
492  timer += Time.deltaTime;
493  // Make bullet path line renderer look more realistic, we will make them stop
494  lineRenderer.SetPosition(0,gunTip.transform.position);
495  lineRenderer.SetPosition (1, gunTip.transform.position);
496  lineRenderer.enabled = true;
497  // Make gun fire look more real
498  lightFromGunFire.enabled = false;
499  setGunPositionWhileFiring (amountOfGunKickBackFiring, false); // Move the gun forward
500  }
501  }
502  else if(ammoDetails.extraClipsLeft > 0)
503  {
504  ownerClientBeganFiring = false;
505  if(isReloadingFinished && isThrowingGrenadeFinished)
506  {
507  ikControl.requestReloading ();
508  playReloadSound ();
509  }
510  isReloadingFinished = false;
511  stopAllComponenets ();
512  }
513  else
514  {
515  ownerClientBeganFiring = false;
516  stopAllComponenets ();
517  }
518  }

696  private void createPhysicsRayCastAndLineRenderer()
697  {
698  Vector3 endPod = new Vector3 (gunTip.transform.localPosition.x + 1000.0f, gunTip.transform.localPosition.y, gunTip.transform.localPosition.z);
699  endPod = gunTip.transform.TransformPoint (endPod);
700  // Create physics ray cast
701  RaycastHit hitInfo;
702  Vector3 direction = endPod - gunTip.transform.position;
703  int layerMask = 1 << 18; // 18 is to ignore barriers
704  layerMask = ~layerMask;
705  Ray ray = new Ray (gunTip.transform.position, direction);
706  Physics.Raycast (ray, out hitInfo, 1000.0f, layerMask);
707  if(hitInfo.collider != null)
708  {
709  endPod = hitInfo.point;
710  // TODO process the ray cast result
711  if(hitInfo.collider.gameObject.tag == "Player")
712  {
713  //Instantiate (bloodSpatter, endPod, Quaternion.identity);
714  if(isUsingMultiplayer)
715  {
716  /*
717  * We will recieve damage only from remote client's fire only when they hit local player.
718  * This procedure is adopted to make fair gameplay even if there is a lag player in room
719  */
720  if(!photonViewComponent.isMine)
721  {
722  if(hitInfo.collider.gameObject.GetComponent<PhotonView>().isMine)
723  {
724  hitInfo.collider.gameObject.GetComponent<SMPlayerHealth> ().reduceHealthPointsBy
725  (ammoDetails.damageMade, photonViewComponent.owner.ID);
726  }
727  }
728  }
729  else
730  {
731  hitInfo.collider.gameObject.GetComponent<SMPlayerHealth> ().reduceHealthPointsBy (ammoDetails.damageMade);
732  }
733  }
734  else if(hitInfo.collider.transform.root.tag == "explode")
735  {
736  Instantiate (wallSpark, endPod, Quaternion.identity);
737  }
738  }
739 
740  // Show the line renderer in bullet path
741  lineRenderer.SetPosition (0, gunTip.transform.position);
742  lineRenderer.SetPosition (1, endPod);
743  lineRenderer.enabled = true;
744  }

783  private void onReloadingFinished()
784  {
785  isReloadingFinished = true;
786  int deficientBullets = noOfBulletsInClip (guntype) - ammoDetails.bulletsLeft;
787  if((totalNumberOfBullets - ammoDetails.bulletsLeft) > deficientBullets)
788  {
789  ammoDetails.bulletsLeft = noOfBulletsInClip (guntype);
790  }
791  else
792  {
793  int allExtraBullets = totalNumberOfBullets - ammoDetails.bulletsLeft;
794  ammoDetails.bulletsLeft += allExtraBullets;
795  }
796  int extraBullets = totalNumberOfBullets - ammoDetails.bulletsLeft;
797  ammoDetails.extraClipsLeft = (extraBullets / noOfBulletsInClip (guntype));
798  if((extraBullets % noOfBulletsInClip(guntype)) != 0)
799  {
800  ammoDetails.extraClipsLeft += 1;
801  }
802 
803  if(weaponInHandComponent != null)
804  {
805  if(audioSource.isPlaying)
806  {
807  audioSource.Stop ();
808  }
809  audioSource.clip = weaponInHandComponent.getGunSound ();
810  }
811  }
