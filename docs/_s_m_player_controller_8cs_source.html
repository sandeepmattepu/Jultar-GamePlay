414  private void checkForSingleTouch()
415  {
416  if(Input.touchCount > 0)
417  {
418  Touch singleTouch = new Touch();
419  int noOfGameTouches = 0;
420  foreach(Touch touch in Input.touches)
421  {
422  if(SMTouchManager.processWhichTouchBelongsTo(touch) == TouchBelongs.GAME)
423  {
424  singleTouch = touch;
425  noOfGameTouches++;
426  }
427  }
428 
429  if(noOfGameTouches == 1) // Check for single tap
430  {
431  if(singleTouch.phase == TouchPhase.Began)
432  {
433  singleTouchTimer += Time.deltaTime;
434  }
435  else if(singleTouch.phase == TouchPhase.Moved || singleTouch.phase == TouchPhase.Stationary)
436  {
437  singleTouchTimer += Time.deltaTime;
438  }
439  else if(singleTouch.phase == TouchPhase.Ended)
440  {
441  if(singleTouchTimer <= 0.30f)
442  {
443  timeAfterSingleTouch += Time.deltaTime;
444  prevIsSingleTap = true;
445  }
446  singleTouchTimer = 0.0f;
447  }
448  }
449  else
450  {
451  singleTouchTimer = 0.0f;
452  }
453  }
454  }

629  private Vector3 rodriguesFormulaToRotateVectors(Vector3 axisOfRot, Vector3 vectToRot, float angleToRotate)
630  {
631  Vector3 unitAxis = (axisOfRot.normalized);
632  Vector3 result = (vectToRot * (Mathf.Cos (angleToRotate * Mathf.Deg2Rad))) + (Vector3.Cross (unitAxis, vectToRot) * Mathf.Sin (angleToRotate * Mathf.Deg2Rad))
633  + (unitAxis * (Vector3.Dot (unitAxis, vectToRot)) * (1 - Mathf.Cos (angleToRotate * Mathf.Deg2Rad)));
634  return result;
635  }

643  private float angleBtwVectorsInDegrees(Vector3 v1, Vector3 v2)
644  {
645  float dotProduct = Vector3.Dot (v1, v2);
646  float angleInRad = Mathf.Acos (dotProduct / ((v1.magnitude) * (v2.magnitude)));
647  return (angleInRad * Mathf.Rad2Deg);
648  }

712  private void setMovementAnimationParameter()
713  {
714  float angleOfOrientStick = Mathf.Atan2(orientationVector.x, orientationVector.y) * Mathf.Rad2Deg; // Unity uses clock-wise notation for angles
715  float angleOfMovementStick = Mathf.Atan2(movementVector.x, movementVector.y) * Mathf.Rad2Deg; // Unity uses clock-wise notation for angles
716  float angleDifference = (angleOfMovementStick - angleOfOrientStick);
717  angleDifference = angleDifference < 0 ? (angleDifference += 360) : angleDifference;
718  /*
719  * Angle 0 starts from top of the circle and angle increases in clockwise direction.
720  * Animations are divided on this circle as Forward at top and ForwardRight, Right, BackwardRight, Backward, BackwardLeft, Left, ForwardLeft in clock-wise direction
721  */
722  Vector3 final = new Vector3();
723  // Converting angle difference to joystick reference angle
724  float angleWithXY = ((-angleDifference) + 90.0f);
725  angleWithXY = angleWithXY < 0 ? (angleWithXY += 360) : angleWithXY;
726  final.x = (movementVector.magnitude * (Mathf.Cos(angleWithXY * Mathf.Deg2Rad)));
727  final.y = (movementVector.magnitude * (Mathf.Sin(angleWithXY * Mathf.Deg2Rad)));
728 
729  animator.SetFloat("HorizontalInput", final.x);
730  animator.SetFloat("HorizontalVertical", final.y);
731  animator.SetFloat("Magnitude", final.magnitude);

791  private void processInputWhileLeaning()

794  if(currentAnimationState != PlayerAnimType.STAND_TO_LEAN && currentAnimationState != PlayerAnimType.LEAN_TO_STAND)

797  float angleOfMovementStick = Mathf.Atan2(movementVector.x , movementVector.y) * Mathf.Rad2Deg; // Unity uses clock-wise notation for angles
798  float angleDifference = (angleOfMovementStick - angleOfOrientStick);
799  if(angleDifference < 0) // Making range of angleDifference as (0 to 360)
800  {
801  angleDifference += 360;
802  }
803  if((angleDifference <= 112.5f && angleDifference > 67.5f) && movementVector.magnitude >= 0.65f)
804  {
805  // Move left

807  {
808  currentFramePlayerRequestAnimState = PlayerAnimType.MOVE_LEFT;

810  else
811  {
812  currentFramePlayerRequestAnimState = PlayerAnimType.TURN_LEFT;
813  isLeaningLeft = true;
814  }
815  }
816  else if(angleDifference <= 292.5f && angleDifference > 247.5f)
817  {
818  // move right

820  {
821  currentFramePlayerRequestAnimState = PlayerAnimType.TURN_RIGHT;
822  isLeaningLeft = false;
823  }
824  else
825  {
826  currentFramePlayerRequestAnimState = PlayerAnimType.MOVE_RIGHT;
827  }
828  }

831  //left idle or right idle

835  }
