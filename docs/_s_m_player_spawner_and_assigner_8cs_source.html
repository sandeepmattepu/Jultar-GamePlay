306  private Transform placePlayerPositionAtStart()
307  {
308  if(gameType == MPGameTypes.FREE_FOR_ALL)
309  {
310  List<int> playerIDs = new List<int> ();
311  int requiredIndex = 0;
312  foreach(PhotonPlayer photonPlayer in PhotonNetwork.playerList)
313  {
314  playerIDs.Add(photonPlayer.ID);
315  }
316  playerIDs.Sort();
317  int numberOfPlayersInEachRegion = playerIDs.Count / 2;
318  for(int i = 0; i < playerIDs.Count; i++)
319  {
320  if(playerIDs[i] == PhotonNetwork.player.ID)
321  {
322  requiredIndex = i;
323  break;
324  }
325  }
326 
327  if(requiredIndex < numberOfPlayersInEachRegion) // Belongs to region one
328  {
329  return region1SpawnPoints [requiredIndex];
330  }
331  else // Belongs to region two
332  {
333  return region2SpawnPoints [requiredIndex - numberOfPlayersInEachRegion];
334  }
335  }
336  else if(gameType == MPGameTypes.TEAM_DEATH_MATCH)
337  {
338  int localPlayerTeam = 1;
339  int requiredIndex = 0;
340  if(SMTeamDeathMatch.PlayerIdAndTeamIndex.TryGetValue (PhotonNetwork.player.ID, out localPlayerTeam))
341  {
342  List<int> teamPlayers = new List<int> ();
343  foreach(KeyValuePair<int,int> vp in SMTeamDeathMatch.PlayerIdAndTeamIndex)
344  {
345  if(vp.Value == localPlayerTeam)
346  {
347  teamPlayers.Add (vp.Key);
348  }
349  }
350  teamPlayers.Sort ();
351  for(int i = 0; i < teamPlayers.Count; i++)
352  {
353  if(teamPlayers[i] == PhotonNetwork.player.ID)
354  {
355  requiredIndex = i;
356  break;
357  }
358  }
359  if(localPlayerTeam == 1)
360  {
361  return region1SpawnPoints [requiredIndex];
362  }
363  else if(localPlayerTeam == 2)
364  {
365  return region2SpawnPoints [requiredIndex];
366  }
367  }
368  else
369  {
370  Debug.LogError ("Recursive function calling");
371  return placePlayerPositionAtStart (); // This wont happen but on safe side
372  }
373  }
374 
375  return this.transform; // this is to avoid error code doesn't reach here
376  }

584  Dictionary<int,int> playerIdAndDeathInfo = new Dictionary<int, int> ();
585 
586  float[] times = {

589  };
590 
591  object[] data = new object[0];
592 
593  if(SMMultiplayerGame.gameType == MPGameTypes.FREE_FOR_ALL)

610  {
611  foreach(KeyValuePair<int,ObscuredInt> pidAS in SMFreeForAll.PlayersIdAndScore)
612  {

620  data = new object[] {
621  times,
