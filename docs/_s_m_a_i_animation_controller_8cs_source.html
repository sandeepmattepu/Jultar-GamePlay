203  private void performFiring()
204  {
205  if(timer >= secondsPerBullet)

207  timer = 0.0f;
208  // play firing audio

211  audioSourceComponent.Stop ();

213  audioSourceComponent.Play ();
214 
215  // Show particle effects
216  if(gunTip.GetComponent<ParticleSystem>().isPlaying)
217  {
218  gunTip.GetComponent<ParticleSystem> ().Stop ();
219  }
220  gunTip.GetComponent<ParticleSystem> ().Play ();
221 
222  // Show line renderer and cast physics raycast
223  createPhysicsRayCastAndLineRenderer();
224  }
225  else

227  timer += Time.deltaTime;
228  // Make bullet path line renderer look more realistic, we will make them stop

260  private void createPhysicsRayCastAndLineRenderer()
261  {
262  Vector3 endPod = new Vector3 (gunTip.transform.localPosition.x + 1000.0f, gunTip.transform.localPosition.y, gunTip.transform.localPosition.z);
263  endPod = gunTip.transform.TransformPoint (endPod);
264  // Create physics ray cast
265  RaycastHit hitInfo;
266  Ray ray = new Ray (gunTip.transform.position, endPod);
267  Physics.Raycast (ray, out hitInfo);
268  if(hitInfo.collider != null)
269  {
270  endPod = hitInfo.point;
271  // TODO process the ray cast result
272  if(hitInfo.collider.gameObject.tag == "Player")
273  {
274  hitInfo.collider.gameObject.GetComponent<SMPlayerHealth> ().reduceHealthPointsBy (ammoDetailsOfGun.damageMade);
275  }
276  }
277 
278  // Show the line renderer in bullet path
279  lineRenderer.SetPosition (0, gunTip.transform.position);
280  lineRenderer.SetPosition (1, endPod);
281  lineRenderer.enabled = true;
282  }
