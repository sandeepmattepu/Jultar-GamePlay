149  private void DragCamera ()
150  {
151  foreach (Touch touch in Input.touches)
152  {
153  if (SMTouchManager.processWhichTouchBelongsTo (touch) == TouchBelongs.GAME) // Drag the camera only when the player touches in non UI region
154  {
155  if (touch.phase == TouchPhase.Began)
156  {
157  lastTouch = touch;
158  touchId = touch.fingerId;
159  }
160  else if (touch.phase == TouchPhase.Moved && touch.fingerId == touchId)
161  {
162  if (lastTouch.position.x - touch.position.x > 0) {
163  // Rotate camera to right
164  angleMadeWithVerticalFromPlayer += 1 * cameraRotationOfVerticalAxisSensitivity;
165  } else if (lastTouch.position.x - touch.position.x < 0) {
166  angleMadeWithVerticalFromPlayer -= 1 * cameraRotationOfVerticalAxisSensitivity;
167  }
168  lastTouch = touch;
169  }
170  }
171  }
172 
173  #if UNITY_EDITOR
174 
175  if(Input.GetMouseButtonDown(0))
176  {
177  previousMousePosition = Input.mousePosition;
178  int layer = touchManager.layerTouchMadeOn(previousMousePosition);
179  mouseClicked = (layer != 5);
180  }
181  else if(Input.GetMouseButtonUp(0))
182  {
183  mouseClicked = false;
184  }
185  else if(mouseClicked)
186  {
187  if(previousMousePosition.x - Input.mousePosition.x > 0)
188  {
189  // Rotate camera to right
190  angleMadeWithVerticalFromPlayer += 1 * cameraRotationOfVerticalAxisSensitivity;
191  }
192  else if(previousMousePosition.x - Input.mousePosition.x < 0)
193  {
194  angleMadeWithVerticalFromPlayer -= 1 * cameraRotationOfVerticalAxisSensitivity;
195  }
196  previousMousePosition = Input.mousePosition;
197  }

205  private void setCameraAppropriately()
206  {
207  Vector3 targetMovementDifference = targetCameraShouldFocus.position - lastPositionOfTarget;
208  if(targetMovementDifference.magnitude >= cameraResponseDelayToPlayerMovement)
209  {
210  lastPositionOfTarget = targetCameraShouldFocus.position;
211  if(targetCameraShouldFocus != this.transform && targetCameraShouldFocus != null)
212  {
213  bool foundNoneBetweenCameraAndPlayer = true;
214  if(cameraAlwaysFocusAtPlayer)
215  {
216  Vector3 directionVector = virtualCameraTransform.position - targetCameraShouldFocus.position;
217  rayFromTargetToCamera = new Ray (targetCameraShouldFocus.position, directionVector);
218  Physics.Raycast (rayFromTargetToCamera, out hitInfo, distanceFromTarget);
219  if(hitInfo.collider != null)
220  {
221  foreach(string tagOfObstructor in cameraAndPlayerObstructingGameObjectTags)
222  {
223  if(hitInfo.collider.tag == tagOfObstructor)
224  {
225  foundNoneBetweenCameraAndPlayer = false;
226  transform.position = Vector3.Lerp(transform.position, hitInfo.point, Time.deltaTime * cameraTransitionSpeed);
227  transform.LookAt (targetCameraShouldFocus);
228  if(meshRenderer != null)
229  {
230  meshRenderer.enabled = true;
231  meshRenderer = null;
232  }
233  break;
234  }
235  }
236  }
237  else if(hitInfo.collider != null && hitInfo.collider.tag == "HideMesh")
238  {
239  foundNoneBetweenCameraAndPlayer = false;
240  meshRenderer = hitInfo.collider.gameObject.GetComponent<MeshRenderer>();
241  if(meshRenderer != null)
242  {
243  meshRenderer.enabled = false;
244  }
245  }
246  }
247 
248  if(!cameraAlwaysFocusAtPlayer || foundNoneBetweenCameraAndPlayer)
249  {
250  cameraOrientation = Quaternion.Euler(angleMadeWithHorizontalFromPlayer, -angleMadeWithVerticalFromPlayer, 0);
251  cameraPosition = cameraOrientation * new Vector3(0, 0, -distanceFromTarget) + targetCameraShouldFocus.position;
252  transform.rotation = Quaternion.Slerp(transform.rotation, cameraOrientation, Time.deltaTime * cameraTransitionSpeed);
253  transform.position = Vector3.Lerp(transform.position, cameraPosition, Time.deltaTime * cameraTransitionSpeed);
254  if (meshRenderer != null)
255  {
256  meshRenderer.enabled = true;
257  meshRenderer = null;
258  }
259  }
260  }
261  }
262  }
