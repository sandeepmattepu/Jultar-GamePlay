368  private void requestForRocketFire(object sender, Vector3 touchPositionOnScreen)
369  {
370  if(rocketIsInAimingPosition)
371  {
372  Ray touchRay;
373  if(Input.touchCount > 0)
374  {
375  touchRay = Camera.main.ScreenPointToRay (touchPositionOnScreen);
376  }
377  else if(Input.GetMouseButtonDown(0))
378  {
379  touchRay = Camera.main.ScreenPointToRay (touchPositionOnScreen);
380  }
381  RaycastHit hitInfo;
382  int layerMask = 1 << 18; // 18 is to ignore barriers collider
383  layerMask |= 1 << 8; // 8 is to ignore player(To avoid rocket flare on player head)
384  layerMask = ~layerMask;
385  if(Physics.Raycast(touchRay, out hitInfo, 500.0f, layerMask))
386  {
387  Vector3 target = hitInfo.point;
388  onCancelButtonPressed ();
389  removeSpecialAbilities(AbilityType.Rocket);
390  rocketIsInAimingPosition = false;
391  rocketButton.interactable = false;
392  rocketButton.GetComponent<Image> ().color = Color.black;
393  PhotonNetwork.RaiseEvent ((byte)MultiplayerEvents.Announcements, (object)AbilityType.Rocket, true, null);
394  makeAnnoucement (friendlyRocketUsedClip);
395  if(jetFlyOverAudio != null)
396  {
397  if(jetFlyOverAudio.isPlaying)
398  {
399  jetFlyOverAudio.Stop ();
400  }
401  jetFlyOverAudio.Play ();
402  }
403  Vector3 rocketInstantiateLocation = target;
404  rocketInstantiateLocation.y += 50.0f;
405  object[] dataToTransfer = new object[1];
406  dataToTransfer [0] = (object)target;
407  GameObject rocketAfterInstantiating = PhotonNetwork.Instantiate (rocketPrefab.name, rocketInstantiateLocation,
408  rocketPrefab.transform.rotation, 0, dataToTransfer) as GameObject;
409  rocketAfterInstantiating.GetComponent<SMRocketBehaviour> ().targetPosition = target;
410  PhotonNetwork.Instantiate (flareForRocketTarget.name, target, flareForRocketTarget.transform.rotation, 0);
411  }
412  }
