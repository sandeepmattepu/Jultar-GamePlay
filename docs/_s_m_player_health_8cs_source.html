296  private void regenerateHealth()
297  {
298  if(playerHealth < MaxHealth)
299  {
300  if(timerForHealthRegenWaiting > secondsToWaitBeforeHealthRegeneration)
301  {
302  if(isUsingMultiPlayer && photonViewComponent.isMine)
303  {
304  playerHealth += ((MaxHealth / secondsToRegenerateFullHealth) * Time.deltaTime);
305  }
306  else if(!isUsingMultiPlayer)
307  {
308  playerHealth += ((MaxHealth / secondsToRegenerateFullHealth) * Time.deltaTime);
309  }
310  if(playerHealth >= MaxHealth)
311  {
312  timerForHealthRegenWaiting = 0.0f;
313  playerHealth = MaxHealth;
314  }
315  setSliderValues ();
316  }
317  else
318  {
319  timerForHealthRegenWaiting += Time.deltaTime;
320  }
321  }

330  private void createDeadBody(bool isMultiplayer, bool canListenAudio, int helmetType)
331  {

333  {
334  if(OnLocalPlayerDead != null)

336  OnLocalPlayerDead ();

338  transform.GetComponent<SMWeaponManager> ().dropHoldingWeaponToFloorWhenDead ();
339  }
340  GameObject ragdollBody = Instantiate(ragdoll, transform.position, transform.rotation) as GameObject;
341  SMDestroyRagdoll destroyRagdoll = ragdollBody.GetComponent<SMDestroyRagdoll>();
342  destroyRagdoll.isMultiplayer = isMultiplayer;
343  destroyRagdoll.canListenToAudio = canListenAudio;
344  ragdollBody.GetComponent<SMPlayerEquipper> ().assignHelmetToDeadBody (helmetType);
345  Camera gameCamera = transform.GetComponent<SMPlayerController>().characterFocusedCamera;
346  if(gameCamera != null)
347  {
348  gameCamera.GetComponent<SMThirdPersonCamera>().targetCameraShouldFocus = ragdollBody.transform;
349  gameCamera.GetComponent<SMThirdPersonCamera>().rotationKnob = ragdollBody.transform;
350  }
351  Transform[] aliveBody = transform.GetComponentsInChildren<Transform>();
352  Transform[] deadBody = ragdollBody.GetComponentsInChildren<Transform>();
353 
354  for (int i = 0; i < aliveBody.Length; i++)
355  {
356  for (int j = 0; j < deadBody.Length; j++)
357  {
358  if (aliveBody[i].gameObject.name == deadBody[j].gameObject.name)
359  {
360  deadBody[j].position = aliveBody[i].position;
361  deadBody[j].rotation = aliveBody[i].rotation;
362  break;
363  }
364  }
365  }
