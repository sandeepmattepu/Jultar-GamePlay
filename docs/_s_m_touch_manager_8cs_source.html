76  private void calculateTouchTypes()
77  {
78  if(isGameTouchMade)

80  // Below if code is written because touches are value types and changes every frame.

85  {

91  }
92  }
93 
94  if((gameTouch.position - initialPositionOfTouch).magnitude < 0.20f)
95  {
96  if(timerAfterTouch < 1.5f)
97  {
98  TouchPhase gameTouchPhase = gameTouch.phase;
99  if(gameTouchPhase == TouchPhase.Canceled || gameTouchPhase == TouchPhase.Ended)
100  {
101  if(timerAfterTouch < 0.30f)
102  {
103  if(OnSingleGameTap != null)
104  {
105  OnSingleGameTap (this, gameTouch.position);
106  }
107  }
108  isGameTouchMade = false;
109  timerAfterTouch = 0.0f;
110  }
111  else
112  {
113  timerAfterTouch += Time.deltaTime;
114  }
115  }
116  else
117  {
118  if(OnGameLongPress != null)
119  {
120  OnGameLongPress (this, gameTouch);
121  }
122  isGameTouchMade = false;
123  timerAfterTouch = 0.0f;
124  }
125  }
126  else
127  {
128  isGameTouchMade = false;
129  timerAfterTouch = 0.0f;
130  }
131  }
132  else if(Input.touchCount > 0)
133  {
134  foreach(Touch touch in Input.touches)
