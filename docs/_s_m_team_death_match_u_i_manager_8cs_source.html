138  }

141  playersTextUI [psui.uiIndex].text = psui.name + " " + psui.score + "/" + psui.deaths;

143  }
144  // Below code is to arrange the position of the scoreboard
145  if(theEffectedPlayer.teamIndex == 1)
146  {
147  team1Players.Sort ();
148  for(int i = 0; i < team1Players.Count; i++)
149  {
150  team1Players [i].uiIndex = i;
151  playersTextUI [i].text = team1Players [i].name + " " + team1Players [i].score + "/" + team1Players [i].deaths;
152  setUIColor (playersTextUI [i], MpPlayerRaceType.Monio, team1Players [i].playerID);
153  }
154  }
155  else if(theEffectedPlayer.teamIndex == 2)
156  {
157  team2Players.Sort ();
158  for(int i = 3; i < (team2Players.Count + 3); i++)
159  {
160  team2Players [i-3].uiIndex = i;

162  setUIColor (playersTextUI [i], MpPlayerRaceType.Jagur, team2Players [i-3].playerID);
163  }
164  }
165  int team1ScoreData = SMTeamDeathMatch.Team1Score;
166  int team2ScoreData = SMTeamDeathMatch.Team2Score;

294  private void onTeamSplitted()
295  {
296  int teamMembersin1 = 0;
297  int teamMembersin2 = 0;
298  foreach(PlayerScoreUI psui in playerScoreAndUI)
299  {
300  foreach(KeyValuePair<int,int> playerIdAndTeamID in SMTeamDeathMatch.PlayerIdAndTeamIndex)
301  {
302  if(playerIdAndTeamID.Key == psui.playerID)
303  {
304  psui.teamIndex = playerIdAndTeamID.Value;
305  psui.uiIndex = -1; // To remove any attachments to UI
306  if(playerIdAndTeamID.Value == 1)
307  {
308  teamMembersin1 += 1;
309  }
310  else if(playerIdAndTeamID.Value == 2)
311  {
312  teamMembersin2 += 1;
313  }
314  break;
315  }
316  }
317  }
318 
319  // code to hide un used ui in team 1
320  int unUsedUILength = Mathf.Abs(3 - teamMembersin1);
321  for(; unUsedUILength > 0; unUsedUILength--)
322  {
323  playersTextUI [3 - unUsedUILength].gameObject.SetActive (false);
324  }
325 
326  // code to hide un used ui in team 2
327  unUsedUILength = Mathf.Abs(3 - teamMembersin2);
328  for(; unUsedUILength > 0; unUsedUILength--)
329  {
330  playersTextUI [6 - unUsedUILength].gameObject.SetActive (false);
331  }
332 
333  if(SMTeamDeathMatch.LocalPlayerTeamIndex == 1)
334  {
335  localPlayerRace = MpPlayerRaceType.Monio;
336  }
337  else if(SMTeamDeathMatch.LocalPlayerTeamIndex == 2)
338  {
339  localPlayerRace = MpPlayerRaceType.Jagur;
340  }
341  setUIColor (monioRaceTotalScore, MpPlayerRaceType.Monio);
342  setUIColor (jagurRaceTotalScore, MpPlayerRaceType.Jagur);
343  monioRaceTotalScore.text = "Monio" + System.Environment.NewLine + SMTeamDeathMatch.Team1Score;
344  jagurRaceTotalScore.text = "Jagur" + System.Environment.NewLine + SMTeamDeathMatch.Team2Score;
345  assignUIIndexToPlayers ();
346  }
347  }
