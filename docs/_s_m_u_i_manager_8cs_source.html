81  {

83  {
84  victoryText.gameObject.SetActive (true);
85  clipTobePlayed = SMTeamDeathMatch.LocalPlayerTeamIndex == 1 ? monioVictorySound : jagurVictorySound;
86  }
87  else

90  clipTobePlayed = SMTeamDeathMatch.LocalPlayerTeamIndex == 1 ? monioDefeatSound : jagurDefeatSound;
